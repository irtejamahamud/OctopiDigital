üìö Introduction to Express.js

Express.js is a minimal and flexible Node.js web application framework that provides a robust set of features for web and mobile applications. It's the most popular Node.js framework and forms the backbone of countless web applications.

üéØ What is Express?

A fast, unopinionated, minimalist web framework for Node.js that provides a thin layer of fundamental web application features.

‚ö° Why Express?

Lightweight, flexible, extensive middleware support, large community, and perfect for building APIs and web applications.

üîß Key Features

Routing, middleware, templating engines, error handling, static file serving, and database integration.

üí° Fun Fact: Express.js was created by TJ Holowaychuk in 2010 and is now maintained by the Node.js Foundation.

üõ£Ô∏è Routing

Routing defines how an application responds to client requests to specific endpoints.

HTTP Methods

// GET request
app.get('/users', (req, res) => {
    res.json({ message: 'Get all users' });
});

// POST request
app.post('/users', (req, res) => {
    res.json({ message: 'Create new user' });
});

// PUT request
app.put('/users/:id', (req, res) => {
    res.json({ message: `Update user ${req.params.id}` });
});

// DELETE request
app.delete('/users/:id', (req, res) => {
    res.json({ message: `Delete user ${req.params.id}` });
});

Route Parameters

// Route parameters
app.get('/users/:id', (req, res) => {
    const userId = req.params.id;
    res.json({ userId });
});

// Multiple parameters
app.get('/users/:id/posts/:postId', (req, res) => {
    const { id, postId } = req.params;
    res.json({ userId: id, postId });
});

// Optional parameters
app.get('/posts/:year/:month?', (req, res) => {
    res.json(req.params);
});

Query Parameters

// Query parameters: /search?q=express&limit=10
app.get('/search', (req, res) => {
    const { q, limit = 20, page = 1 } = req.query;
    res.json({
        query: q,
        limit: parseInt(limit),
        page: parseInt(page)
    });
});

Express Router

// routes/users.js
const express = require('express');
const router = express.Router();

// Middleware for this router
router.use((req, res, next) => {
    console.log('User routes middleware');
    next();
});

router.get('/', (req, res) => {
    res.json({ users: [] });
});

router.post('/', (req, res) => {
    res.json({ message: 'User created' });
});

module.exports = router;

// app.js - Using the router
const userRoutes = require('./routes/users');
app.use('/api/users', userRoutes);

üîß Middleware

Middleware functions execute during the request-response cycle and can modify request/response objects, end the request-response cycle, or call the next middleware.

Built-in Middleware

// Body parser middleware
app.use(express.json()); // Parse JSON bodies
app.use(express.urlencoded({ extended: true })); // Parse URL-encoded bodies

// Static files middleware
app.use(express.static('public')); // Serve static files

// CORS middleware
app.use((req, res, next) => {
    res.header('Access-Control-Allow-Origin', '*');
    res.header('Access-Control-Allow-Headers', 'Content-Type');
    next();
});

Custom Middleware

// Logging middleware
const logger = (req, res, next) => {
    const timestamp = new Date().toISOString();
    console.log(`${timestamp} - ${req.method} ${req.url}`);
    next(); // Important: call next() to continue
};

// Authentication middleware
const authenticate = (req, res, next) => {
    const token = req.headers.authorization;
    
    if (!token) {
        return res.status(401).json({ error: 'No token provided' });
    }
    
    // Verify token logic here
    req.user = { id: 1, name: 'John Doe' }; // Decoded token data
    next();
};

// Apply middleware
app.use(logger);
app.use('/api/protected', authenticate);
